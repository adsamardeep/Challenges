class Solution:
    def getDecimalValue(self, head: ListNode) -> int:
        num = 0
        length = 0

        tmp = head
        while tmp:
            length += 1
            tmp = tmp.next

        tmp = head
        while tmp:
            length -= 1
            if tmp.val == 1:
                num += (1 << length)
            tmp = tmp.next

        return num
